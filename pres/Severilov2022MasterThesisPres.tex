\begin{frame}{Введение}
	\textbf{Рассматривается задача} выбора оптимальной модели декодирования сигналов. 
	Процесс декодирования заключается в восстановлении зависимости между двумя наборами данных. \\ \vspace{0.3cm}
	
	\textbf{Проблема}: пространства исходных и целевых переменных обладают существенно избыточной размерностью.\\ \vspace{0.3cm}

	\textbf{Гипотеза}: построение отображения из исходного пространства в целевое как отображения между бесконечномерными пространствами. \\ \vspace{0.3cm}
	
	%Актуальность: предполагается, что рассматриваемый подход к декодированию сигналов применим к сигналам с носимых устройств:
	%1 для передискретизации сигнала при низких частотах семплирования
	%2 для обработки нерегулярных по времени данных при непостоянном контакте со считывающим устройством
	
	\textbf{Метод решения:}
	Предлагаются подходы нейронного оператора и нейронного Фурье оператора.
	
\end{frame}

\begin{frame}{Постановка задачи}
	\begin{itemize}
		\item $\mathcal{X}=\mathcal{X}\left(D ; \mathbb{R}^{d_{x}}\right)$ и $\mathcal{Y}=\mathcal{Y}\left(D ; \mathbb{R}^{d_{y}}\right)$ -- банаховы пространства функций на $D \subset \mathbb{R}^{x}$ 
		\item $\Psi: \mathcal{X} \rightarrow \mathcal{Y}$ -- нелинейное отображение
		\item Последовательность наблюдений $\left\{x_{j}, y_{j}\right\}_{j=1}^{N}$, где $y_{j}=\Psi\left(x_{j}\right)$ 
		\item Требуется построить аппроксимацию $\Psi$
		$$
		\Psi_{\theta}: \mathcal{X} \rightarrow \mathcal{Y}, \quad \theta \in \Theta, \quad \quad \quad \Psi_{\theta^{\dagger}} \approx \Psi, \quad \theta^{\dagger} \in \Theta
		,$$
		где $\Theta$ -- конечномерное пространство параметров
		\item Задача оптимизации
		$$
		\min_{\theta \in \Theta} \mathbb{E}_{x}\left[C\left(\Psi_{\theta}(x), \Psi(x)\right)\right]
		,$$
		где $C: \mathcal{Y} \times \mathcal{Y} \rightarrow \mathbb{R}$ -- функция ошибки
	\end{itemize}

\begin{frame}{Нейронный оператор}
	Итеративная структура решения:
	$$x\mapsto v_{0} \mapsto v_{1} \mapsto \ldots \mapsto v_{K} \mapsto y,$$
	где $v_{j}, j=0,1, \ldots, K-1$ -последовательность функций, $v_{0}(p)=P(x(t))$,  $x(t)=Q\left(v_{K}(t)\right), Q: \mathbb{R}^{d_{v}} \rightarrow \mathbb{R}^{d_{y}}$, $P(.), Q(.)$ -- полносвязные слои
	\begin{block}{Определение 1 (Слой нейронного оператора)}
		$$v_{k+1}(t):=\sigma\left(W v_{k}(t)+\left(\mathcal{K}(x ; \phi) v_{k}\right)(t)\right), \quad \forall t \in D,$$
		где  $\mathcal{K}(x ; \phi)$ -- ядерный интегральный оператор, $W: \mathbb{R}^{d_{v}} \rightarrow \mathbb{R}^{d_{v}}$ -- линейное преобразование, $\sigma: \mathbb{R} \rightarrow \mathbb{R}$ -- нелинейная функция активации
	\end{block}
	\begin{block}{Определение 2 (Ядерный интегральный оператор)}
		$$
		\left(\mathcal{K}(x ; \phi) v_{k}\right)(t):=\int_{D} \kappa(t, q, x(t), x(q) ; \phi) v_{k}(q) \mathrm{d} q, \quad \forall p \in D,$$
		где  $\kappa_{\phi}: \mathbb{R}^{2\left(d+d_{x}\right)} \rightarrow \mathbb{R}^{d_{v} \times d_{v}}$ --  нейросеть, параметризованная $\phi \in \Theta_{\mathcal{K}}$
		
	\end{block}
	
	
\end{frame}
%-----------------------------------------------------------------------------------------------------
\begin{frame}{Нейронный Фурье оператор}

	
	\begin{block}{Определение 3 (Интегральный Фурье оператор)}
		Убирая зависимость от функции $x$ и применяя теорему о свертке, получим:
		$$
		\left(\mathcal{K}(\phi) v_{k}\right)(t)=\mathcal{F}^{-1}\left(R_{\phi} \cdot\left(\mathcal{F} v_{k}\right)\right)(t) \quad \forall t \in D,
		$$ 
		где $R_{\phi} = \mathcal{F}\left(\kappa_{\phi}\right) $ -- преобразование Фурье функции $\kappa: \bar{D} \rightarrow \mathbb{R}^{d_{v} \times d_{v}}$ параметризованной $\phi \in \Theta_{\mathcal{K}}$
		
	\end{block}

\begin{frame}{Понижение размерностей}
	 \textbf{Гипотеза}: 
	 \begin{enumerate}
	 	\item Понижение размерностей исходного и целевого пространств на основе методов PCA, PLS (отображения $F_{\mathcal{X}}, F_{\mathcal{Y}}$ в размерность ниже, отображения в исходную размерность $G_{\mathcal{X}}, G_{\mathcal{Y}}$)
	 	\item Построение отображения $\chi$ между пространствами с размерностью ниже исходной
	 \end{enumerate}
 $$
 \begin{aligned}
 G_{\mathcal{X}} \circ F_{\mathcal{X}} & \approx I_{\mathcal{X}} \\
 G_{\mathcal{Y}} \circ F_{\mathcal{Y}} & \approx I_{\mathcal{Y}} \\
 \end{aligned}$$
 $$\boxed{G_{\mathcal{Y}} \circ \chi \circ F_{\mathcal{X}}  \approx \Psi} $$
